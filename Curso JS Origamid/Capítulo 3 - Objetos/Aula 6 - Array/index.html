Arrays sempre começam endereçadas por 0

const instrumentos = ['Guitarra', 'Baixo', 'Violão'];
const precos = [49, 99, 69, 89];

Saída: array armazenada

const dados = [new String('Tipo 1'), ['Carro', 'Portas', {cor:'Azul', preco: 2000}], function andar(nome) { console.log(nome)}];

Saída: array armazenada

dados[2]('Ford');

Saída: Ford

dados[1][2].cor;

const carros = new Array('Ford', 'Fiat', 'Honda');

Array.of(), Array() e new Array()

Array.of(10);

Saída: [10]

Array.of sempre criará o conteúdo indicado como uma array

Array.of(1,2,3,4);

Saída: [1, 2, 3, 4]

new Array(5);

Saída: [empty × 5]

Uma new Array, quando indicado um número somente, ira criar uma array vazia com o length total do número indicado

Array(5);

Saída: [empty × 5]

O mesmo ocorre quando feita apenas como Array

Array(1,2,3,4);

Saída: [1, 2, 3, 4]

Porém quando se separa com vírgulas, criar uma array dessa forma, criará diversas array's com os números indicados

.unshift() e .push()

const carros2 = ['Ford', 'Fiat', 'VW'];

Saída: ['Ford', 'Fiat', 'VW']

console.log(carros2);

Saída: ['Ford', 'Fiat', 'VW']

const novaArray = carros2.unshift('Honda', 'Kia');

Saída: novaArray = 5

console.log(novaArray);

Saída: 5

console.log(carros2);

Saída: ['Honda', 'Kia', 'Ford', 'Fiat', 'VW']

const novaArray2 = carros2.push('Ferrari', 'Toyota', 'Subaru');

Saída: novaArray2 = 8

console.log(novaArray2);

Saída: 8

console.log(carros2);

Saída: ['Honda', 'Kia', 'Ford', 'Fiat', 'VW', 'Ferrari', 'Toyota', 'Subaru']

.shift() e .pop()

const primeiroCarro = carros2.shift();

Saída: primeiroCarro = 'Honda'

console.log(primeiroCarro);

Saída: 'Honda'

console.log(carros2);

Saída: ['Kia', 'Ford', 'Fiat', 'VW', 'Ferrari', 'Toyota', 'Subaru']

const ultimoCarro = carros2.pop();

Saída: ultimoCarro = 'Subaru'

console.log(ultimoCarro);

Saída: 'Subaru'

console.log(carros2);

Saída: ['Kia', 'Ford', 'Fiat', 'VW', 'Ferrari', 'Toyota']

carros2.reverse();

Saída: ['Toyota', 'Ferrari', 'VW', 'Fiat', 'Ford', 'Kia']

console.log(carros2);

Saída: ['Toyota', 'Ferrari', 'VW', 'Fiat', 'Ford', 'Kia']

.splice(index, remover, item1, item2, ...)

const carros3 = ['Ford', 'Fiat', 'VW', 'Honda'];

Saída: ['Ford', 'Fiat', 'VW', 'Honda']

carros3.splice(1, 0, 'Kia', 'Mustang');

Saída: ['Ford', 'Kia', 'Mustang', 'Fiat', 'VW', 'Honda']

console.log(carros3);

Saída: ['Ford', 'Kia', 'Mustang', 'Fiat', 'VW', 'Honda']

carros3.splice(3, 2, 'Ferrari');

Saída: ['Ford', 'Kia', 'Mustang', 'Ferrari', 'Honda']

console.log(carros3);

Saída: ['Ford', 'Kia', 'Mustang', 'Ferrari', 'Honda']
