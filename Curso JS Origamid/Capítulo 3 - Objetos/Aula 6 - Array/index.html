Aula 6 - Array

Array's [0]

Arrays sempre começam endereçadas por 0

const instrumentos = ['Guitarra', 'Baixo', 'Violão'];
const precos = [49, 99, 69, 89];

Saída: array armazenada

const dados = [new String('Tipo 1'), ['Carro', 'Portas', {cor:'Azul', preco: 2000}], function andar(nome) { console.log(nome)}];

Saída: array armazenada

dados[2]('Ford');

Saída: Ford

dados[1][2].cor;

Saída: 'Azul'

const carros = new Array('Ford', 'Fiat', 'Honda');

Saída: carros = ['Ford', 'Fiat', 'Honda']

carros[2] = 'Ferrari';

Saída: carros = ['Ford', 'Fiat', 'Ferrari']

carros[3] = 'Kia';

Saída: carros = ['Ford', 'Fiat', 'Ferrari', 'Kia']

carros[20] = 'Kia';

Saída: carros[20] = 'Kia' (índices 4 a 19 ficam vazios)

console.log(carros.length);

Saída: 21

let li = document.querySelectorAll('li');

Saída: NodeList de <li> do documento

li = Array.from(li);

Saída: li agora é um Array

const carros2 = { 0: 'Fiat', 1: 'Honda', 2: 'Ford', length: 4 };

Saída: carros2 é um objeto parecido com array

const carrosArray = Array.from(carros2);

Saída: carrosArray = ['Fiat', 'Honda', 'Ford', undefined]
